def matxMultiply(A, B):
    
    if len(A[0]) != len(B) :
        raise ValueError
    
    result = [[0] * len(B[0]) for i in range(len(A))]
    for i in range(len(A)):
        for j in range(len(B[0])):
          for k in range(len(B)):
            result[i][j] += A[i][k] * B[k][j]                   
    return result

def gj_Solve(A,b,decPts=4,epsilon=1.0e-16):
    if len(A) != len(b):
        raise ValueError
 
    Ab = augmentMatrix(A,b)
 
    for j in range(len(Ab[0])-1): #
        maxValue = 0
        row = 0
        for i in range(len(Ab)):
            if i >= j:
                if abs(Ab[i][j])>maxValue:
                    maxValue = abs(Ab[i][j])
                    row = i
                    
        if abs(maxValue) < epsilon:
            return None #
        
        #
        if (row != j):
            swapRows(Ab,j,row)
            row = 0
            maxValue = 0
        
        scaleRow(Ab,j,1/Ab[j][j])
            
        for x in range(len(A)):
            if Ab[x][j] != 0 and x != j:
                addScaledRow(Ab,x,j,-Ab[x][j])
                
        new_Ab = [] #
        
        for i in range(len(Ab)):
            row = []
            row.append(Ab[i][len(Ab)])
            new_Ab.append(row)
            
    return new_Ab

def calculateMSE2D(X,Y,m,b):
    items = []
    for i in range(len(X)):
        value = pow((Y[i] - m*X[i] -b),2)
        items.append(value)
        
    totalValue = sum(items)
    MSE = totalValue / len(X)
    
    return MSE
